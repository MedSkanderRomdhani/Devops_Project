pipeline {
    agent any

    stages {
         stage('Checkout') {
            steps {
                script {
                    // Make sure to use the correct branch name
                    git branch: 'Oussema', url: 'https://github.com/MedSkanderRomdhani/Devops_Project.git'
                }
            }
        }

        stage('Build and Test') {
            steps {
                sh 'mvn clean test'
            }
        }

    }

    post {
        success {
            echo 'Build and tests succeeded BROOOOO !'
        }
        failure {
            echo 'Build or tests failed!'
        }
    }
}
